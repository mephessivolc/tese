\section{Algoritmo de ForĂ§a Bruta}

O algoritmo de ForĂ§a Bruta apresentado no \autoref{alg:brute_force} ĂŠ utilizado ao longo de todo o projeto referente a algoritmo classico de forĂ§a bruta.


\begin{algorithm}
\caption{Algoritmo de ForĂ§a Bruta}
    \label{alg:brute_force}
    \begin{algorithmic}[1]
    \Require matriz de distĂ˘ncias \texttt{dist}
    \Ensure melhor custo e melhor caminho
    \State $n \gets \text{len}(\texttt{dist})$
    \State \texttt{vertices $\gets$ [1, \dots, n-1]}
    \State \texttt{best\_cost $\gets$ inf}
    \State \texttt{best\_path $\gets$ None}
    \ForAll{\texttt{perm} em \texttt{permutations(vertices)}}
        \State \texttt{path $\gets$ (0,) + perm + (0,)}
        \State \texttt{cost $\gets$ 0}
        \For{$i \gets 0$ atĂŠ $\text{len(path)} - 2$}
            \State \texttt{cost $\gets$ cost + dist[path[i]][path[i+1]]}
        \EndFor
        \If{\texttt{cost < best\_cost}}
            \State \texttt{best\_cost $\gets$ cost}
            \State \texttt{best\_path $\gets$ path}
        \EndIf
    \EndFor
    
    \State \Return \texttt{best\_cost, best\_path}
    \end{algorithmic}
\end{algorithm}

\begin{algorithm}[htbp]
\caption{ConstruĂ§ĂŁo do Hamiltoniano do TSP em qudits}

\begin{algorithmic}[1]
\Require matriz de distĂ˘ncias $d_{ij}$, penalizaĂ§ĂŁo $\lambda$, dimensĂŁo local $\mathrm{dim}$
\Ensure Hamiltoniano $H$

\State $N \gets$ nĂşmero de linhas de $d_{ij}$

    \State $\mathrm{dim} \gets N$
\EndIf

    \State lanĂ§ar erro
\EndIf

\For{$k \gets 0$ atĂŠ $N-1$}
    \For{$a \gets 0$ atĂŠ $N-1$}
        \State $P_{k,a} \gets \Call{ProjetorLocal}{k,a,N,\mathrm{dim}}$
    \EndFor
\EndFor

\State $H_{\mathrm{dist}} \gets 0$
\For{$k \gets 0$ atĂŠ $N-1$}
    \State $k' \gets (k+1)\bmod N$
    \For{$a \gets 0$ atĂŠ $N-1$}
        \For{$b \gets 0$ atĂŠ $N-1$}
            \State $H_{\mathrm{dist}} \gets H_{\mathrm{dist}} + d_{ij}[a,b]\,P_{k,a}P_{k',b}$

\EndFor

    \For{$l \gets k+1$ atĂŠ $N-1$}
        \For{$a \gets 0$ atĂŠ $N-1$}
            \State $H_{\mathrm{pen}} \gets H_{\mathrm{pen}} + P_{k,a}P_{l,a}$
        \EndFor
    \EndFor
\EndFor

\end{algorithmic}
\end{algorithm}

\begin{algorithm}[htbp]

\begin{algorithmic}[1]
\Require Hamiltoniano $H$, matriz de distĂ˘ncias $\mathrm{dist}$, nĂşmero de cidades $N$, dimensĂŁo local $\mathrm{dim}$
